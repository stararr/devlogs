local tooltips = {} local mouse = game.Players.LocalPlayer:GetMouse() local tooltipactive = false local paddingl = script.Parent:WaitForChild("UIPadding").PaddingLeft.Offset local paddingr = script.Parent:WaitForChild("UIPadding").PaddingRight.Offset for _, guiitem in pairs(game.Players.LocalPlayer.PlayerGui:GetDescendants()) do if guiitem:GetAttribute("ToolTip") then guiitem.MouseEnter:Connect(function() if #tooltips==0 then script.Parent.Visible = true end if not (table.find(tooltips, guiitem)) then table.insert(tooltips, guiitem) end script.Parent.Text = guiitem:GetAttribute("ToolTip") script.Parent.Position = UDim2.new(0, mouse.X, 0, mouse.Y-12) script.Parent.Size = UDim2.new(0, script.Parent.TextBounds.X+paddingl+paddingr, 0, 18) end) guiitem.MouseMoved:Connect(function() if #tooltips~=0 then script.Parent.Position = UDim2.new(0, mouse.X, 0, mouse.Y-12) end end) guiitem.MouseLeave:Connect(function() table.remove(tooltips, table.find(tooltips, guiitem)) if #tooltips==0 then script.Parent.Visible = false end end) end end
